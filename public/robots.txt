# Public surfaces are crawlable. The CMS surface and API endpoints
# are excluded — they're authenticated and serve no SEO value, and
# crawler hits to /api/* would just generate cold-start invocations
# on the Vercel function. Login page is excluded for the same reason.

Allow: /
Disallow: /admin/
Disallow: /admin
Disallow: /api/
Disallow: /login
